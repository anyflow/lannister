17  package net.anyflow.lannister.packetreceiver;
18  
19  import java.util.Date;
20  
21  import io.netty.channel.ChannelFutureListener;
22  import io.netty.channel.ChannelHandlerContext;
23  import io.netty.channel.SimpleChannelInboundHandler;
24  import io.netty.handler.codec.mqtt.MqttPubAckMessage;
25  import net.anyflow.lannister.AbnormalDisconnectEventArgs;
26  import net.anyflow.lannister.message.OutboundMessageStatus;
27  import net.anyflow.lannister.plugin.DeliveredEventArgs;
28  import net.anyflow.lannister.plugin.DeliveredEventListener;
29  import net.anyflow.lannister.plugin.DisconnectEventListener;
30  import net.anyflow.lannister.plugin.Plugins;
31  import net.anyflow.lannister.session.Session;
32  import net.anyflow.lannister.topic.Topic;
33  import net.anyflow.lannister.topic.TopicSubscriber;
34  import net.anyflow.lannister.topic.Topics.ClientType;
35  
36  public class PubAckReceiver extends SimpleChannelInboundHandler<MqttPubAckMessage> {
37  
38  	private static final org.slf4j.Logger logger = org.slf4j.LoggerFactory.getLogger(PubAckReceiver.class);
39  
40  	@Override
41  	protected void channelRead0(ChannelHandlerContext ctx, MqttPubAckMessage msg) throws Exception {
42  		logger.debug("packet incoming [message={}]", msg.toString());
43  
44  		Session session = Session.NEXUS.get(ctx.channel().id());
45  		if (session == null) {
46  			logger.error("None exist session message [message={}]", msg.toString());
47  
48  			ctx.channel().disconnect().addListener(ChannelFutureListener.CLOSE).addListener(fs -> // [MQTT-4.8.0-1]
49  			Plugins.INSTANCE.get(DisconnectEventListener.class).disconnected(new AbnormalDisconnectEventArgs()));
50  			return;
51  		}
52  
53  		session.setLastIncomingTime(new Date());
54  
55  		String clientId = session.clientId();
56  		int messageId = msg.variableHeader().messageId();
57  
58  		Topic topic = Topic.NEXUS.get(clientId, messageId, ClientType.SUBSCRIBER);
59  		if (topic == null) {
60  			logger.error("Topic does not exist [clientId={}, messageId={}]", clientId, messageId);
61  			session.dispose(true); // [MQTT-3.3.5-2]
62  			return;
63  		}
64  
65  		final TopicSubscriber topicSubscriber = topic.subscribers().get(clientId);
66  
67  		OutboundMessageStatus status = topicSubscriber.outboundMessageStatuses().get(messageId);
68  
69  		if (status != null) {
70  			Plugins.INSTANCE.get(DeliveredEventListener.class).delivered(new DeliveredEventArgs() {
71  				@Override
72  				public String clientId() {
73  					return clientId;
74  				}
75  
76  				@Override
77  				public int messageId() {
78  					return messageId;
79  				}
80  			});
81  		}
82  
83  		topicSubscriber.removeOutboundMessageStatus(messageId);
84  		logger.debug("Outbound message status REMOVED [clientId={}, messageId={}]", clientId, messageId);
85  	}
86  }
